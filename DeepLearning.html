Deep Learning

Pacotes:

library("MASS")
library("rpart")
set.seed(0)

Vamos tentar prever o valor media de uma casa baseados das features disponveis no dataset: Boston Housing Values in Suburbs of Boston

data = Boston
data[is.na(data) == TRUE]
## numeric(0)
head(data)

Separando entre treinamento e teste:

quebra = 0.8 * nrow(data)
train <- data.frame(data[1:quebra,])
test <- data.frame(data[(1+quebra): nrow(data),])

rvore de deciso (anova para regresso e class para classificao):

arvore <- rpart(medv ~.,method="anova", data=train)
tree_predict <- predict(arvore,test)
mse_tree <- mean((tree_predict - test$medv)^2)
mse_tree
## [1] 45.93991

Normalizando segundo x-min/(max-min), para diminuir a intensidade dos rudos:

set.seed(0)
maior_de_cada_coluna = apply(data, 2, max) 
menor_de_cada_coluna = apply(data, 2, min)
diff = maior_de_cada_coluna - menor_de_cada_coluna
scaled <- scale(data, center = menor_de_cada_coluna, scale = diff)

Separando treinamento e teste de forma aleatria:

set.seed(0)
index = sample(1:nrow(data),round(0.70*nrow(data)))
train_data <- as.data.frame(scaled[index,])
test_data <- as.data.frame(scaled[-index,])

Rede neural:

library(neuralnet)
nn <- neuralnet(medv~.,data=train_data,hidden=c(5,4,3,2),linear.output=T)
plot(nn)

hidden: vector of integers specifying the number of hidden neurons (vertices) in each layer. hidden=c(5,4,3,2):

  • Na primeira camada escondida 5 neurnios
  • Na segunda camada escondida 4 neurnios
  • Na terceira camada escondida 3 neurnios
  • Na quarta camada escondida 2 neurnios

Previso com a funo compute (e no com predict) e temos que desfazer o scale para compararmos com a varivel original:

pr.nn <- compute(nn,test_data[,1:13])
pr.nn_ <- pr.nn$net.result*(max(data$medv)-min(data$medv))+min(data$medv)
test.r <- (test_data$medv)*(max(data$medv)-min(data$medv))+min(data$medv)

Calculando erro quadrtico mdio:

MSE_nn <- mean((pr.nn_ - test.r)^2)
MSE_nn
## [1] 9.975509

Plot do observado versus predito:

plot(test_data$medv,type = 'l',col="red",xlab = "x", ylab = "Valor Residencia")
lines(pr.nn$net.result,col = "blue")

## exemplo de rede neural com classificao

Vamos prever a varivel Private, que indica se a universidade pblica ou privada.

library(ISLR)
library(neuralnet)
data <- College
private <- ifelse(data$Private == 'Yes', 1, 0)

Padronizar dados para melhor performance

data <- data[,2:18]
max_data <- apply(data, 2, max) 
min_data <- apply(data, 2, min)
scaled <- data.frame(scale(data,center = min_data, scale = max_data - min_data))

Inclui varivel explicada (target)

scaled$Private <- private

Separando treinamento e teste:

set.seed(0)
index = sample(1:nrow(data),round(0.70*nrow(data)))
train_data <- as.data.frame(scaled[index,])
test_data <- as.data.frame(scaled[-index,])

Modelo rede neural com a opo linear.output=F pois agora estamos no contexto da classificao:

set.seed(0)
nn <- neuralnet(Private ~ .,data=train_data,hidden=c(5,4),linear.output=F)
plot(nn)

Previso:

pr.nn <- compute(nn,test_data[,1:17])
pr.nn$net.result <- sapply(pr.nn$net.result,round,digits=0) # aqui seria melhor usar cutoff e curva ROC
